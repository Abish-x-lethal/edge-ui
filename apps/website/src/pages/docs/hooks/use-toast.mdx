# useToast

This hook can be used to manage toast alerts.

# Signature

```ts
interface ToasterAction {
    toast: ({ ...props }: Toast) => {
        id: string;
        dismiss: () => void;
        update: (props: ToasterToast) => void;
    };
    dismiss: (toastId?: string) => void;
    toasts: ToasterToast[];
}

function useToast(): ToasterAction;
```

### Example

```jsx
import { useToast, Toaster } from '@edge-ui/react';

export function Component() {
    const { toast } = useToast();

    const copyText = () => {
        navigator.clipboard.writeText('Hello World!').then(
            () => {
                toast({
                    title: 'Copied to clipboard!',
                    description: 'The text has been copied to your clipboard!'
                });
            },
            () => {
                toast({
                    title: 'ERROR!',
                    variant: 'destructive',
                    description: 'Failed to copy the text to your clipboard!'
                });
            }
        );
    };

    return (
        <EdgeUIProvider>
            <Toaster />
            <button onClick={toast}>Copy Text</button>
        </EdgeUIProvider>
    );
}
```
