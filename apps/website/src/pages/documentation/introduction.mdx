# Introduction

EdgeUI is a minimal react components library based on [shadcn ui](https://ui.shadcn.com).

# Installation

```pwsh
$ npm install @edge-ui/react
```

# Example

```jsx
// main component
import { EdgeUIProvider } from '@edge-ui/react';

export default function Application({ Component, pageProps }: AppProps) {
    return (
        <EdgeUIProvider>
            <Component {...pageProps} />
        </EdgeUIProvider>
    );
}
```

```jsx
// other components
import { Heading, Button } from '@edge-ui/react';

export function MyComponent() {
    const [count, setCount] = useState(0);

    const counter = () => {
        setCount((p) => p + 1);
    };

    return (
        <div>
            <Heading>Count: {count}</Heading>
            <Button onClick={counter}>Click Me!</Button>
        </div>
    );
}
```
